% We are using the 'authoryear' style in this example.
% The default is the 'numerical' style.
\usepackage{csquotes}

\usepackage[%
  citestyle=authoryear,%
  bibstyle=authortitle,%
  backend=biber,%
  maxcitenames=1,%
  maxbibnames=99,%
  sorting=nty,%
  date=comp,%
  urldate=iso8601,%
  dashed=false,
  uniquename=false,
  uniquelist=false,
  autocite=inline,%% behavior of autocite
  isbn=false,
  doi=false,
  url=false
]{biblatex}

\addbibresource{myBib.bib}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Get always lastname -firstname in the bibligraphy %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\DeclareNameAlias{sortname}{first-last}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Square bracket for cite %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\makeatletter
\newrobustcmd*{\parentexttrack}[1]{%
  \begingroup
  \blx@blxinit
  \blx@setsfcodes
  \blx@bibopenparen#1\blx@bibcloseparen
  \endgroup}
\AtEveryCite{%
  \let\parentext=\parentexttrack%
  \let\bibopenparen=\bibopenbracket%
  \let\bibcloseparen=\bibclosebracket}
\makeatother

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Coma between author and year %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\renewcommand*{\nameyeardelim}{\addcomma\space}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Add [authoryear] at the beginning of the entry in the bibliography %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\newcounter{mymaxcitenames}
\AtBeginDocument{%
  \setcounter{mymaxcitenames}{\value{maxnames}}%
}
\renewbibmacro*{begentry}{%
  \printtext[brackets]{%
    \begingroup
    \defcounter{maxnames}{\value{mymaxcitenames}}%
    \printnames{labelname}%
    \setunit{\nameyeardelim}%
    \printfield{labelyear}%
    \printfield{extrayear}%
    \endgroup
    }%
  \quad% or \addspace
}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% To get et al in italics %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\usepackage{xpatch}
\xpatchbibmacro{name:andothers}{%
  \bibstring{andothers}%
}{%
  \bibstring[\emph]{andothers}%
}{}{}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Let's make some space between entries %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\setlength\bibitemsep{1.5\itemsep}
